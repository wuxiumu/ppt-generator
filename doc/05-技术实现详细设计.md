### 核心模块设计

#### 5.1 CLI入口 - main.py

```python
"""
用法：
  python main.py input.json                          # 输出到 output/<topic>.pptx
  python main.py input.json -o custom.pptx           # 指定输出路径
  python main.py input.json --mode beauty            # 美观极致版
  python main.py input.json --mode balanced          # 平衡版
  python main.py input.json --no-images              # 不生成图片
"""

import argparse
import json
import asyncio
from src.planner import Planner
from src.writer import Writer
from src.image_gen import ImageGenerator
from src.renderer import Renderer
from src.validator import NarrativeValidator
from src.cost_tracker import CostTracker

async def generate_pptx(input_path: str, output_path: str, mode: str = "beauty"):
    # 加载输入
    with open(input_path) as f:
        input_data = json.load(f)
    
    budget = 10.0 if mode == "beauty" else 1.0
    tracker = CostTracker(budget_usd=budget)
    
    # Stage 1: 规划
    planner = Planner(mode=mode, tracker=tracker)
    plan = await planner.plan(input_data)
    # plan 包含: narrative_arc, slide_plan[], style_spec, image_prompts[]
    
    # 校验规划
    validator = NarrativeValidator()
    issues = validator.validate_plan(plan)
    if issues:
        plan = await planner.revise(plan, issues)
    
    # Stage 2: 并行生成内容
    writer = Writer(mode=mode, tracker=tracker)
    slide_contents = await writer.generate_all(plan)
    
    # Stage 2.5: 并行生成图片（美观版）
    images = {}
    if mode == "beauty" and plan.get("image_prompts"):
        image_gen = ImageGenerator(mode=mode, tracker=tracker)
        images = await image_gen.generate_all(plan["image_prompts"])
    
    # 内容校验
    issues = validator.validate_contents(slide_contents, plan)
    if issues:
        # 对有问题的slide重新生成
        slide_contents = await writer.revise(slide_contents, issues, plan)
    
    # Stage 3: 渲染
    renderer = Renderer()
    pptx_path = renderer.render(
        slide_contents=slide_contents,
        style_spec=plan["style_spec"],
        images=images,
        output_path=output_path
    )
    
    # 输出统计
    tracker.print_summary()
    print(f"✅ 生成完成: {pptx_path}")
    print(f"   共 {len(slide_contents)} 张幻灯片")
    print(f"   总耗时: {tracker.elapsed_seconds:.1f}s")
    print(f"   总成本: ${tracker.spent:.4f}")
```

**Planner Prompt 模板的关键设计**：

```
你是一个PPT叙事架构师。你的任务是为一个25-30张的PPT设计完整的叙事结构。

输入：
- 主题：{{ topic }}
- 简介：{{ brief }}
- 受众：{{ audience }}

你需要做以下决策：

1. 叙事类型判断
   根据主题和简介，判断这是哪种叙事：
   - tutorial（教程）：渐进式，概念→示例→练习
   - timeline（时间线）：按时间展开，每段有主题
   - guide（指南）：决策树式，需求→选项→推荐
   - pitch（说服）：问题-方案式，痛点→方案→证据
   - itinerary（行程）：行程式，按时间/地点展开

2. 调色板选择
   从 [tech, warm, corporate, personal, minimal, dark] 中选择

3. 每张slide的规划
   为每张slide指定：
   - narrative_role: 这张slide在叙事中的角色
   - layout_type: 从12种布局中选择
   - content_brief: 50-100字的内容要求
   - transition: 与前后slide的衔接说明

4. 布局节奏控制
   - 连续不超过2张纯文本slide
   - 每4-5张必须有一个视觉重的布局（全图/图表/对比/大数字）
   - 章节分隔页间距不超过6张
   - 封面和结束页要有特殊设计

5. 图片需求
   为需要配图的slide生成DALL-E prompt
```

#### 5.3 Writer - 内容生成器

```python
class Writer:
    def __init__(self, mode="beauty", tracker=None):
        self.mode = mode
        self.tracker = tracker
        # 美观版用更强的模型
        self.model = "claude-3-5-sonnet" if mode == "beauty" else "gpt-4o-mini"
        self.template = load_template("writer_prompt.j2")
        self.semaphore = asyncio.Semaphore(5 if mode == "beauty" else 10)
    
    async def generate_all(self, plan: Plan) -> list[SlideContent]:
        """并行生成所有slide的内容"""
        tasks = []
        for i, slide_spec in enumerate(plan.slide_plan):
            # 获取前后slide的摘要（用于过渡）
            prev_brief = plan.slide_plan[i-1].content_brief if i > 0 else None
            next_brief = plan.slide_plan[i+1].content_brief if i < len(plan.slide_plan)-1 else None
            
            tasks.append(self._generate_one(slide_spec, plan.style_spec, prev_brief, next_brief))
        
        results = await asyncio.gather(*tasks)
        return results
    
    async def _generate_one(self, slide_spec, style_spec, prev_brief, next_brief):
        async with self.semaphore:
            prompt = self.template.render(
                slide=slide_spec,
                style=style_spec,
                prev_context=prev_brief,
                next_context=next_brief
            )
            response = await call_llm(self.model, prompt, response_format="json")
            self.tracker.record(...)
            return SlideContent.from_dict(response)
```

**Writer Prompt 的关键设计**：

```
你是一个PPT内容撰稿人。请为以下slide撰写内容。

当前slide信息：
- 编号：{{ slide.slide_num }}
- 叙事角色：{{ slide.narrative_role }}
- 布局类型：{{ slide.layout_type }}
- 内容要求：{{ slide.content_brief }}
- 过渡说明：从"{{ prev_context }}"过渡到本slide，然后引向"{{ next_context }}"

风格要求：
- 语气：{{ style.tone }}
- 语言：中文
- 视觉隐喻：{{ style.visual_metaphor }}

输出格式（JSON）：
{
    "title": "标题（不超过15字）",
    "subtitle": "副标题（可选）",
    "bullets": ["要点1", "要点2", ...],  // 最多5个
    "body_text": "正文段落（可选，不超过80字）",
    "highlight": "核心数字或关键词（可选）",
    "chart_data": { ... },  // 如果需要图表
    "speaker_notes": "演讲备注（50-100字）"
}

约束：
- 文字要精炼，每张slide不超过80字（标题+要点+正文）
- 不要使用emoji
- 要点之间要有逻辑递进，不是并列罗列

- 如果layout_type是"code_block"，输出代码示例和逐行注释
```

#### 5.4 Renderer - PPTX渲染引擎

```python
from pptx import Presentation
from pptx.util import Inches, Pt, Emu
from pptx.dml.color import RGBColor
from pptx.enum.text import PP_ALIGN

class Renderer:
    def __init__(self):
        self.layouts = load_yaml("layouts.yaml")
        self.palettes = load_yaml("palettes.yaml")
        self.slide_width = Inches(13.333)  # 16:9 widescreen
        self.slide_height = Inches(7.5)
    
    def render(self, slide_contents, style_spec, images, output_path):
        prs = Presentation()
        prs.slide_width = self.slide_width
        prs.slide_height = self.slide_height
        
        palette = self.palettes[style_spec["palette"]]
        
        for content in slide_contents:
            layout = self.layouts[content.layout_type]
            slide = prs.slides.add_slide(prs.slide_layouts[6])  # 空白布局
            
            # 填充背景色
            self._fill_background(slide, layout, palette)
            
            # 根据布局类型渲染
            renderer_method = getattr(self, f"_render_{content.layout_type}")
            renderer_method(slide, content, layout, palette, images)
            
            # 添加页码（封面和结束页除外）
            if content.layout_type not in ("title", "end"):
                self._add_page_number(slide, content.slide_num, len(slide_contents), palette)
        
        prs.save(output_path)
        return output_path
    
    def _render_bullets(self, slide, content, layout, palette, images):
        """渲染要点列表页"""
        # 左侧装饰条
        accent_bar = layout["accent_bar"]
        shape = slide.shapes.add_shape(
            MSO_SHAPE.RECTANGLE,
            Inches(accent_bar["pos"][0] * 13.333),
            Inches(accent_bar["pos"][1] * 7.5),
            Inches(accent_bar["size"][0] * 13.333),
            Inches(accent_bar["size"][1] * 7.5)
        )
        shape.fill.solid()
        shape.fill.fore_color.rgb = self._color(palette, accent_bar["fill"])
        
        # 标题
        self._add_textbox(slide, content.title, layout["title"], palette, "title")
        
        # 要点列表
        items_spec = layout["items"]
        txBox = slide.shapes.add_textbox(...)
        tf = txBox.text_frame
        tf.word_wrap = True
        
        for i, bullet in enumerate(content.bullets):
            p = tf.paragraphs[0] if i == 0 else tf.add_paragraph()
            p.text = bullet
            p.font.size = Pt(18)
            p.font.color.rgb = self._color(palette, "primary_text")
            p.space_after = Pt(12)
            # 添加圆点图标
            p.level = 0
    
    def _render_section(self, slide, content, layout, palette, images):
        """渲染章节分隔页"""
        # 整页背景色（accent_1）
        self._fill_background(slide, {"bg_fill": "accent_1"}, palette)
        
        # 大章节编号
        self._add_textbox(slide, f"0{content.act}", layout["section_number"], palette, "accent_number", text_color="bg")
        
        # 章节标题（白字）
        self._add_textbox(slide, content.title, layout["section_title"], palette, "title", text_color="bg")
        
        # 副标题
        if content.subtitle:
            self._add_textbox(slide, content.subtitle, layout["section_subtitle"], palette, "caption", text_color="bg")
    
    def _render_code_block(self, slide, content, layout, palette, images):
        """渲染代码展示页"""
        # 标题
        self._add_textbox(slide, content.title, layout["title"], palette, "title")
        
        # 代码区域（深色背景）
        code_spec = layout["code_area"]
        code_shape = slide.shapes.add_shape(
            MSO_SHAPE.ROUNDED_RECTANGLE, ...
        )
        code_shape.fill.solid()
        code_shape.fill.fore_color.rgb = RGBColor(0x1E, 0x1E, 0x1E)
        
        # 代码文本（等宽字体）
        code_tf = code_shape.text_frame
        code_tf.text = content.code_text
        for paragraph in code_tf.paragraphs:
            paragraph.font.name = "Consolas"
            paragraph.font.size = Pt(14)
            paragraph.font.color.rgb = RGBColor(0xD4, 0xD4, 0xD4)
        
        # 右侧注释
        self._add_textbox(slide, content.annotations, layout["annotations"], palette, "body")
```

#### 5.5 叙事校验器

```python
class NarrativeValidator:
    def validate_plan(self, plan: Plan) -> list[str]:
        """校验叙事规划的结构合理性"""
        issues = []
        
        # 1. 检查slide数量
        n = len(plan.slide_plan)
        if n < 25 or n > 30:
            issues.append(f"Slide数量 {n} 不在25-30范围内")
        
        # 2. 检查三幕比例
        act_counts = Counter(s.act for s in plan.slide_plan)
        total = sum(act_counts.values())
        act1_ratio = act_counts[1] / total
        act2_ratio = act_counts[2] / total
        act3_ratio = act_counts[3] / total
        
        if act1_ratio < 0.1 or act1_ratio > 0.25:
            issues.append(f"Act 1 比例 {act1_ratio:.0%} 不合理（建议10-20%）")
        if act3_ratio < 0.1 or act3_ratio > 0.25:
            issues.append(f"Act 3 比例 {act3_ratio:.0%} 不合理（建议10-20%）")
        
        # 3. 检查布局连续性
        consecutive_same = 0
        for i in range(1, len(plan.slide_plan)):
            if plan.slide_plan[i].layout_type == plan.slide_plan[i-1].layout_type:
                consecutive_same += 1
                if consecutive_same >= 2:
                    issues.append(f"S{i+1}: 连续{consecutive_same+1}张相同布局({plan.slide_plan[i].layout_type})")
            else:
                consecutive_same = 0
        
        # 4. 检查章节分隔间距
        last_section = 0
        for i, s in enumerate(plan.slide_plan):
            if s.layout_type == "section":
                gap = i - last_section
                if gap > 7:
                    issues.append(f"S{i+1}: 距上一个章节分隔页{gap}张，间隔太大")
                last_section = i
        
        return issues
    
    def validate_contents(self, contents: list[SlideContent], plan: Plan) -> list[str]:
        """校验生成内容的质量"""
        issues = []
        
        for i, (content, spec) in enumerate(zip(contents, plan.slide_plan)):
            # 文字量检查
            total_chars = len(content.title) + sum(len(b) for b in content.bullets) + len(content.body_text or "")
            if total_chars > 150 and spec.layout_type not in ("code_block",):
                issues.append(f"S{i+1}: 文字过多({total_chars}字)，需要精简")
            
            if total_chars < 5 and spec.layout_type not in ("full_image",):
                issues.append(f"S{i+1}: 文字过少({total_chars}字)，内容可能不够")
            
            # 标题长度检查
            if len(content.title) > 20:
                issues.append(f"S{i+1}: 标题过长({len(content.title)}字)，建议15字以内")
        
        return issues
```

### 运行流程（时序）

```
T+0s    读取 input.json
T+1s    调用 Planner LLM（约15-30秒）
T+30s   收到规划，开始校验
T+31s   并行启动 28 个 Writer 调用
T+31s   并行启动 N 个 Image 调用（如有）
T+60s   Writer 陆续返回（GPT-4o-mini 每张约3-5秒）
T+90s   Image 陆续返回（DALL-E 3 每张约10-15秒）
T+95s   所有内容就绪，启动渲染
T+105s  渲染完成，保存 .pptx
T+106s  输出统计信息
```

美观版总耗时约 100-120 秒（< 2分钟），远低于30分钟限制。
平衡版总耗时约 30-45 秒。
